// Fire away!
function execute_request( $args ) {
	global $base_url;
	$target_url = add_query_arg( $args, $base_url );
	$data       = wp_remote_get( $target_url );
	echo '<pre><code>';
	print_r( $data['body'] );
	echo '<code></pre>';
}
